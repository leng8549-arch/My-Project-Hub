⚙️ ตั้งค่า Apps Script URL (ทำครั้งเดียว)

วาง Web App URL จาก Apps Script ที่ Deploy แล้ว → ดูคู่มือขั้นตอน

1. เปิด script.google.com → สร้างโปรเจกต์ใหม่
2. วางโค้ดนี้ทั้งหมด:

function doGet(e){var p=e.parameter,id=p.id,mode=p.mode||'files';if(mode==='folders'){var parent=DriveApp.getFolderById(id);var result=[];var it=parent.getFolders();while(it.hasNext()){var f=it.next();var imgs=[];var fi=f.getFiles();while(fi.hasNext()){var file=fi.next();imgs.push('https://drive.google.com/thumbnail?id='+file.getId()+'&sz=w800');}result.push({id:f.getId(),name:f.getName(),images:imgs,date:''});}return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON);}var folder=DriveApp.getFolderById(id);var urls=[];var fit=folder.getFiles();while(fit.hasNext()){var file=fit.next();urls.push('https://drive.google.com/thumbnail?id='+file.getId()+'&sz=w800');}return ContentService.createTextOutput(JSON.stringify(urls)).setMimeType(ContentService.MimeType.JSON);}

3. Deploy → Web app · Execute as: Me · Access: Anyone
4. Copy URL มาวางในช่องด้านบน
